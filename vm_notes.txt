   FROM       auth_users           u
   INNER JOIN auth_group_users     gu ON (gu.UserId      =  u.UserId)
   INNER JOIN auth_dataset_userset du ON (du.UserGroupId IN (u.UserId,gu.GroupId))
   LEFT  JOIN auth_groups          g  ON (g.GroupId      =  du.UserGroupId)
   INNER JOIN auth_datasets        d  ON (d.DatasetId    =  du.DatasetId)
   WHERE (u.UserId=1);
